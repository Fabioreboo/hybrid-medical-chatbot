

● 🏥 Medical Chatbot with User Authentication & Admin Console - Implementation 
  Plan

  Executive Summary

  Transform the existing medical chatbot into a multi-user web application with
  role-based access control, Google OAuth authentication, and a sophisticated
  knowledge base management system. This plan provides a phased approach to
  implement user accounts, admin controls, and enhanced data governance while
  maintaining the core chatbot functionality.

  1.2 Authentication Flow

  sequenceDiagram
      participant U as User
      participant FE as Frontend
      participant BE as Backend
      participant Google as Google OAuth
      participant DB as Database

      U->>FE: Click "Login with Google"
      FE->>Google: Redirect to OAuth
      Google->>U: Authenticate
      U->>Google: Grant permission
      Google->>BE: Return ID token
      BE->>Google: Verify token
      Google->>BE: Token verification OK
      BE->>DB: Check if user exists
      alt User exists
          DB->>BE: Return user record
      else New user
          BE->>DB: Create user record
          DB->>BE: Return new record
      end
      BE->>FE: Return JWT session
      FE->>U: Grant access

  1.3 Data Isolation Rules

  - Chat Messages: user_id foreign key, filtered at query level
  - Saved Entries: user_id + is_public (admin only)
  - KB Entries: created_by (user_id), approved_by (admin_id), version
  - Audit Logs: user_id, action, timestamp, affected_resource

  const ensureDataIsolation = (req, res, next) => {

  4.2 Database Schema Design

  -- Users table
  CREATE TABLE users (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      google_id VARCHAR(255) UNIQUE NOT NULL,
      email VARCHAR(255) UNIQUE NOT NULL,
      name VARCHAR(255) NOT NULL,
      role ENUM('user', 'admin') DEFAULT 'user',
      created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      is_active BOOLEAN DEFAULT true
  );

  -- Chat Messages
  CREATE TABLE chat_messages (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      user_id UUID REFERENCES users(id) ON DELETE CASCADE,
      message TEXT NOT NULL,
      response TEXT NOT NULL,
      source ENUM('database', 'llm_fallback') NOT NULL,
      created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      is_saved BOOLEAN DEFAULT false
  );

  -- Knowledge Base
  CREATE TABLE knowledge_base (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      symptom VARCHAR(255) NOT NULL,
      drug VARCHAR(255) NOT NULL,
      mechanism TEXT,
      precautions TEXT,
      side_effects TEXT,
      created_by UUID REFERENCES users(id),
      approved_by UUID REFERENCES users(id),
      status ENUM('pending', 'approved', 'rejected') DEFAULT 'pending',
      version INTEGER DEFAULT 1,
      created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
      updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
  );

  -- Audit Logs
  CREATE TABLE audit_logs (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      user_id UUID REFERENCES users(id),
      action VARCHAR(100) NOT NULL,
      resource_type VARCHAR(50),
      resource_id UUID,
      metadata JSONB,
      ip_address INET,
      created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
  );
